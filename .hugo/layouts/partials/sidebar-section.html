{{- /* Renders one content section as a sidebar nav item, recursing into any subsections. Sections with children become a collapsible
/ block; leaf sections are a plain link. Params: "page" (the section to render), "current" (the page being viewed). */ -}} {{- $page := .page -}} {{- $current := .current -}} {{- $children := $page.Sections.ByTitle -}} {{- $isActive := or (eq $page $current) ($current.IsDescendant $page) -}} {{- if $children }} {{ $page.Title }}
    {{- range $children }} {{- partial "sidebar-section.html" (dict "page" . "current" $current) }} {{- end }}
{{- else }} {{ $page.Title }} {{- end }}
